def draw (ages, cases=cases, fig = None, label=' '):
    if fig is None:
        fig, ax = plt.subplots()
    ax = fig.get_axes()[0]
    ax.hist(cases.Age, range=(0,100), bins=ages, weights=cases.Weight, label=label)
    ax.set_xticks(ages)
    ax.set_xticklabels(['{:.2f}'.format(a) for a in ages], rotation=90)
    ax.legend()
    return fig, ax
