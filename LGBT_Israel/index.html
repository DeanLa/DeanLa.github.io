big = pd.read_excel('./lgbti.xlsx').drop(0, axis=0)
fb = big[['Post ID', 'Posted', 'Lifetime Post Total Reach','Lifetime Engaged Users','Type']]
fb.columns = [['ID', 'Posted', 'Reach','Engaged', 'Type']]
fb['Reach'] = fb.Reach.astype('int')
fb['Engaged'] = fb.Engaged.astype('int')
fb['Posted'] = fb['Posted'] + pd.Timedelta(hours=10)
fb['Prev'] = fb.Posted.shift(-1)
fb.drop([265,266], axis=0, inplace=True)
fb['Gap'] = (fb.Posted - fb.Prev).astype('timedelta64[m]') / 60.
fb['Control'] = (fb.Posted < np.datetime64('2015-02-01'))
fb['TOD'] = fb.Posted.map(lambda x: x.hour + x.minute/60.)
fb['Date'] = fb.Posted.map(lambda x: x.date())
fb['DOW'] = fb.Posted.map(lambda x: (x.dayofweek + 1) % 7 + 1)
print "Sample:"
fb.sample(4)
